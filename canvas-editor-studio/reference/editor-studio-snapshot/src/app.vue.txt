function toPlainHex(color: string | undefined, fallback: string) {
    if (!color) return fallback;
    if (/^#[0-9a-fA-F]{6}$/.test(color)) return color;
    if (/^#[0-9a-fA-F]{8}$/.test(color)) return color.slice(0, 7);

    const matched = color.match(/\d+(\.\d+)?/g);
    if (!matched || matched.length < 3) return fallback;

    const [r, g, b] = matched.slice(0, 3).map((item) => Number(item));

    return `#${[r, g, b]
        .map((value) => Math.max(0, Math.min(255, value)).toString(16).padStart(2, '0'))
        .join('')}`;
}

function buildBoardTemplet() {
    const templet = cloneDeep(boardTemplet as Record<string, any>);
    const page = templet.pages[0];
    const layout = page.elements[0];

    templet.global.layout.width = STAGE_WIDTH;
    templet.global.layout.height = STAGE_HEIGHT;
    templet.global.layout.backgroundColor = '#101927ff';
    templet.global.layout.backgroundImage = null;
    templet.global.layout.backgroundSize = [0, 0];
    templet.global.layout.backgroundImageInfo = {
        opacity: 1,
        width: STAGE_WIDTH,
        height: STAGE_HEIGHT,
        transform: { a: 1, b: 0, c: 0, d: 1, tx: 0, ty: 0 },
    };

    layout.width = STAGE_WIDTH;
    layout.height = STAGE_HEIGHT;
    layout.background = {
        color: '#101927ff',
        watermarkEnable: false,
        image: null,
    };
    layout.backgroundColor = '#101927ff';
    layout.backgroundGradient = null;
    layout.backgroundMask = {
        color: '#000000ff',
        opacity: 0,
    };
    layout.elements = [];

    return templet;
}

function buildTextContents(
    element: TextElementModel | null,
    overrides: Partial<TextContent> & { content?: string },
) {
    const baseContent =
        element?.contents?.[0] ||
        ({
            content: element?.content || '',
            fontSize: element?.fontSize,
            fontFamily: element?.fontFamily,
            fontWeight: element?.fontWeight,
            color: element?.color,
        } as TextContent);

    return [
        {
            ...baseContent,
            ...overrides,
        },
    ];
}

function markDirty() {
    isSceneDirty.value = true;
}

function setSelectionToScene() {
    if (!editor.value) return;
    editor.value.focusElement(null);
}

function focusScene() {
    setSelectionToScene();
}

function clearLayoutElements(layout: LayoutModel) {
    if (!editor.value) return;
    const existing = layout.elements.slice();
    if (existing.length) {
        editor.value.removeElement(existing, layout);
    }
}

async function toObjectUrl(dataUrl: string) {
    const response = await fetch(dataUrl);
    const blob = await response.blob();

    return URL.createObjectURL(blob);
}

async function resolvePresetAssets() {
    const resolved = await Promise.all(
        SCENE_PRESETS.map(async (preset) => [
            preset.id,
            {
                backgroundUrl: await toObjectUrl(preset.backgroundUrl),
                heroUrl: await toObjectUrl(preset.heroUrl),
            },
        ]),
    );

    presetAssetUrls.value = Object.fromEntries(resolved) as Record<
        ScenePresetId,
        { backgroundUrl: string; heroUrl: string }
    >;
}

function addSceneText(
    layout: LayoutModel,
    config: {
        content: string;
        left: number;
        top: number;
        width?: number;
        fontSize: number;
        fontFamily: string;
        fontWeight: number;
        color: string;
        textAlign?: 'left' | 'center' | 'right';
        lineHeight?: number;
    },
) {
    if (!editor.value) return;

    editor.value.addText(
        config.content,
        {
            left: config.left,
            top: config.top,
            width: config.width,
            fontSize: config.fontSize,
            fontFamily: config.fontFamily,
            fontWeight: config.fontWeight,
            color: config.color,
            textAlign: config.textAlign || 'left',
            lineHeight: config.lineHeight || 1.08,
            letterSpacing: 0,
        } as Partial<BaseElementModel>,
        layout,
    );
}

function setBackgroundImage(
    layout: LayoutModel,
    options: {
        url: string;
        naturalWidth: number;
        naturalHeight: number;
        color: string;
        fitMode: 'cover' | 'contain';
        opacity: number;
    },
) {
    if (!editor.value) return;

    const ratioBase =
        options.fitMode === 'contain'
            ? Math.min(layout.width / options.naturalWidth, layout.height / options.naturalHeight)
            : Math.max(layout.width / options.naturalWidth, layout.height / options.naturalHeight);

    const backgroundImage = {
        ...(layout.background?.image || {}),
        url: options.url,
        width: layout.width,
        height: layout.height,
        left: 0,
        top: 0,
        opacity: options.opacity,
        resourceType: 'image',
        naturalWidth: options.naturalWidth,
        naturalHeight: options.naturalHeight,
        transform: parseTransform({ a: 1, b: 0, c: 0, d: 1, tx: 0, ty: 0 }),
        imageTransform: parseTransform({
            a: ratioBase,
            b: 0,
            c: 0,
            d: ratioBase,
            tx: 0,
            ty: 0,
        }),
        filterInfo: {
            id: '',
            url: '',
            strong: 0.8,
            intensity: 0.8,
            zipUrl: '',
        },
    };

    editor.value.changeLayout(
        {
            background: {
                ...(layout.background || { watermarkEnable: false }),
                color: options.color,
                image: backgroundImage,
            },
        },
        layout,
    );
}

function applyScenePreset(presetId: ScenePresetId) {
    if (!editor.value || !currentLayout.value) return;

    selectedPresetId.value = presetId;
    const preset = SCENE_PRESETS.find((item) => item.id === presetId) || SCENE_PRESETS[0];
    const assets = presetAssetUrls.value[preset.id];
    const layout = currentLayout.value;

    clearLayoutElements(layout);
    setBackgroundImage(layout, {
        url: assets?.backgroundUrl || preset.backgroundUrl,
        naturalWidth: preset.backgroundNaturalWidth,
        naturalHeight: preset.backgroundNaturalHeight,
        color: preset.backgroundColor,
        fitMode: 'cover',
        opacity: 1,
    });

    editor.value.addElement(
        {
            type: 'image',
            left: preset.heroFrame.left,
            top: preset.heroFrame.top,
            width: preset.heroFrame.width,
            height: preset.heroFrame.height,
            url: assets?.heroUrl || preset.heroUrl,
            naturalWidth: preset.heroNaturalWidth,
            naturalHeight: preset.heroNaturalHeight,
            resourceType: 'image',
            opacity: 1,
        },
        layout,
    );

    addSceneText(layout, {
        content: preset.label,
        left: 96,
        top: 96,
        fontSize: 18,
        fontFamily: DEFAULT_FONT,
        fontWeight: 600,
        color: preset.accentColor,
    });
    addSceneText(layout, {
        content: preset.headline,
        left: 96,
        top: 154,
        width: 520,
        fontSize: 74,
        fontFamily: DEFAULT_FONT,
        fontWeight: 700,
        color: preset.textColor,
        lineHeight: 1.02,
    });
    addSceneText(layout, {
        content: preset.body,
        left: 100,
        top: 466,
        width: 460,
        fontSize: 24,
        fontFamily: DEFAULT_BODY_FONT,
        fontWeight: 400,
        color: preset.textColor,
        lineHeight: 1.36,
    });
    addSceneText(layout, {
        content: preset.footer,
        left: 98,
        top: 720,
        fontSize: 18,
        fontFamily: DEFAULT_BODY_FONT,
        fontWeight: 500,
        color: preset.accentColor,
    });

    backgroundColorInput.value = toPlainHex(preset.backgroundColor, '#101927');
    backgroundOpacityInput.value = 100;
    backgroundFitMode.value = 'cover';
    isSceneDirty.value = false;
    setSelectionToScene();
    editor.value.fitZoom();
}

function updateTextElement(
    patch: Partial<TextElementModel>,
    contentPatch?: Partial<TextContent> & { content?: string },
) {
    if (!editor.value || !isTextSelection.value || currentElement.value?.type !== 'text') return;
    const target = currentElement.value as TextElementModel;
    const contents = buildTextContents(target, contentPatch || {});

    editor.value.changeElement(
        {
            ...patch,
            contents,
        },
        target,
    );
    markDirty();
}

function updateTextContent(event: Event) {
    const value = (event.target as HTMLTextAreaElement).value;
    textDraft.value = value;
    updateTextElement(
        {
            content: value,
        } as Partial<TextElementModel>,
        { content: value },
    );
}

    } as Partial<TextElementModel>);
}

function addTextBlock(kind: 'headline' | 'body') {
    if (!editor.value || !currentLayout.value) return;

    const content = kind === 'headline' ? '新的主标题' : '新的说明文案，继续在这里细化卖点与语气。';
    const fontSize = kind === 'headline' ? 64 : 24;
    const fontFamily = kind === 'headline' ? DEFAULT_FONT : DEFAULT_BODY_FONT;
    const fontWeight = kind === 'headline' ? 700 : 400;
    const color = kind === 'headline' ? currentPreset.value.textColor : currentPreset.value.textColor;
    const left = kind === 'headline' ? 160 : 180;
    const top = kind === 'headline' ? 220 : 560;

    const element = editor.value.addText(
        content,
        {
            left,
            top,
            fontSize,
            fontFamily,
            fontWeight,
            color,
            textAlign: 'left',
            lineHeight: kind === 'headline' ? 1.02 : 1.36,
        } as Partial<BaseElementModel>,
        currentLayout.value,
    );

    editor.value.focusElement(element);
    markDirty();
}

async function uploadBackground() {
    const file = await pickFile({
        accept: 'image/png,image/jpeg,image/jpg,image/webp,image/svg+xml',
    });
    if (!file || !currentLayout.value) return;

    const url = URL.createObjectURL(file);
    const image = await loadImage(url);
    setBackgroundImage(currentLayout.value, {
        url,
        naturalWidth: image.naturalWidth,
        naturalHeight: image.naturalHeight,
        color: `${backgroundColorInput.value}ff`,
        fitMode: backgroundFitMode.value,
        opacity: backgroundOpacityInput.value / 100,
    });
    markDirty();
}

async function replaceSelectedImage() {
    if (!editor.value || !isImageSelection.value || !currentElement.value) return;

    const file = await pickFile({
        accept: 'image/png,image/jpeg,image/jpg,image/webp,image/svg+xml',
    });
    if (!file) return;

    const url = URL.createObjectURL(file);
    const image = await loadImage(url);

    editor.value.changeElement(
        {
            url,
            naturalWidth: image.naturalWidth,
            naturalHeight: image.naturalHeight,
        },
        currentElement.value,
    );
    markDirty();
}

function updateBackgroundColor(event: Event) {
    const value = (event.target as HTMLInputElement).value;
    backgroundColorInput.value = value;
    if (!editor.value || !currentLayout.value) return;

    editor.value.changeLayout(
        {
            background: {
                ...(currentLayout.value.background || { watermarkEnable: false }),
                color: `${value}ff`,
            },
        },
        currentLayout.value,
    );
    markDirty();
}

function updateBackgroundOpacity(event: Event) {
    const value = Number((event.target as HTMLInputElement).value);
    backgroundOpacityInput.value = value;
    if (!editor.value || !currentLayout.value?.background?.image) return;

    editor.value.changeLayout(
        {
            background: {
                ...(currentLayout.value.background || { watermarkEnable: false }),
                image: {
                    ...currentLayout.value.background.image,
                    opacity: value / 100,
                },
            },
        },
        currentLayout.value,
    );
    markDirty();
}

function changeBackgroundFit(mode: 'cover' | 'contain') {
    backgroundFitMode.value = mode;
    if (!currentLayout.value?.background?.image) return;

    const image = currentLayout.value.background.image;
    setBackgroundImage(currentLayout.value, {
        url: image.url,
        naturalWidth: image.naturalWidth,
        naturalHeight: image.naturalHeight,
        color: currentLayout.value.background?.color || `${backgroundColorInput.value}ff`,
        fitMode: mode,
        opacity: image.opacity || 1,
    });
    markDirty();
}

function clearBackground() {
    if (!editor.value || !currentLayout.value) return;

    editor.value.changeLayout(
        {
            background: {
                ...(currentLayout.value.background || { watermarkEnable: false }),
                image: null,
            },
        },
        currentLayout.value,
    );
    markDirty();
}

function duplicateSelection() {
    if (!editor.value || !currentElement.value) return;

    const [copy] = editor.value.copyElement(currentElement.value);
    copy.left = currentElement.value.left + 36;
    copy.top = currentElement.value.top + 36;
    const [pasted] = editor.value.pasteElement(copy);
    editor.value.focusElement(pasted);
    markDirty();
}

function removeSelection() {
    if (!editor.value || !currentElement.value || !currentElement.value.parent) return;

    editor.value.removeElement(currentElement.value, currentElement.value.parent);
    setSelectionToScene();
    markDirty();
}

function layerName(layer: BaseElementModel) {
    if (layer.type === 'text') {
        const content = (layer as TextElementModel).contents?.[0]?.content || layer.content || '文字';
        return content.replace(/\n/g, ' ').slice(0, 18);
    }
    if (layer.type === 'image') {
        return '主视觉图片';
    }
    return layer.type;
}

function focusLayer(layer: BaseElementModel) {
    if (!editor.value) return;
    editor.value.focusElement(layer);
}

function undo() {
    editor.value?.undo();
}

function redo() {
    editor.value?.redo();
}

function exportPng() {
    if (!editor.value) return;

    editor.value.plugins.invokeCommand(
        'infiniteCore:export',
        currentLayout.value,
        'png',
        () => undefined,
        (result: Blob | null) => {
            if (!result) return;

            const url = URL.createObjectURL(result);
            const link = document.createElement('a');
            link.download = `editor-studio-${selectedPresetId.value}.png`;
            link.href = url;
            link.click();
            URL.revokeObjectURL(url);
        },
    );
}

watch(
    currentElement,
    (element) => {
        if (!element || element.type !== 'text') return;

        const textElement = element as TextElementModel;
        const content = textElement.contents?.[0];
        textDraft.value = content?.content || textElement.content || '';
        fontFamilyDraft.value = content?.fontFamily || textElement.fontFamily || DEFAULT_FONT;
        fontSizeDraft.value = Number(content?.fontSize || textElement.fontSize || 72);
        fontWeightDraft.value = Number(content?.fontWeight || textElement.fontWeight || 700);
        textColorDraft.value = toPlainHex(content?.color || textElement.color, '#f8f2e8');
        textAlignDraft.value = (textElement.textAlign as 'left' | 'center' | 'right') || 'left';
    },
    { immediate: true },
);

watch(
    currentLayout,
    (layout) => {
        if (!layout) return;

        backgroundColorInput.value = toPlainHex(layout.background?.color, '#101927');
        backgroundOpacityInput.value = Math.round((layout.background?.image?.opacity || 1) * 100);
    },
    { immediate: true },
);

onMounted(async () => {
    editorOptions.value = {
        fontList: fontListJson as unknown as Font[],
        mode: 'board',
    };

    await nextTick();

    if (!editorRef.value) return;

    await editorRef.value.setTemplet(buildBoardTemplet());
    await resolvePresetAssets();
    editorRef.value.plugins.invokeCommand('guide:toggleVisible', true);
    editorRef.value.plugins.invokeCommand('ruler:toggleVisible', true);
    applyScenePreset(selectedPresetId.value);

    // @ts-expect-error 调试入口
    window.editorStudio = editorRef.value;
});
